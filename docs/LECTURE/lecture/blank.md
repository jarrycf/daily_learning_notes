```
public static void Reverse(LinkListClass<Interger> L){
    LinkNode<Integer> p=L.head.next, q  // p指向首结点
    L.head.next = null; // 将L置为一个空表
    while(p!=null){
        q=p.next;  // q临时保存p结点的后继结点
        p.next=L.head.next; // 将p结点插入到表头
        L.head.next = p;
        p=q;
    }
}
```
